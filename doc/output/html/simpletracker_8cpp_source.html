53 void SimpleTracker::processKilobot(Kilobot_detection_library::Kilobot &kilobot, int pathLimit) {
54 bool found = false;
55 for (auto &[rtl, k, path]: kilobots) {
56 auto diff = k->box & kilobot.box;
57 if (((float) diff.area() / ((float) (k->box.area() + kilobot.box.area()) / 2)) > 0.6) {
58 k->box = kilobot.box;
59 if (rtl > 0) rtl = 0;
60
61 auto last = path.back();
62 if (auto centerDiff = last - kilobot.center(); (abs(centerDiff.x) + abs(centerDiff.y)) > 2) {
63 path.push_back(kilobot.center());
64 if ((pathLimit > 0) && (path.size() > pathLimit)) path.pop_front();
65 }
66
67 found = true;
