169 YoloDetector::_parseNNResults(const std::vector<std::vector<Mat>> &nnOutput, std::vector<float> &confidences,
170 std::vector<cv::Rect> &boxes, int resultRows, float xScale,
171 float yScale, float scoreVal, float confVal) {
172 //iterate over the data
173 for (const auto &layer: nnOutput) {
174 for (const auto &result: layer) {
175 for (int i = 0; i < resultRows; ++i) {
176
177 // [0, 1, 2, 3, 4, 5]
178 // [x, y, width, height, confidence, score]
179 struct tPredict {
180 float x;
181 float y;
182 float width;
183 float height;
184 float confidence;
185 float score;
186 };
187 static_assert(sizeof(tPredict) == (sizeof(float) * 6));
188
189 tPredict prediction = result.at<tPredict>(i);
190 if ((prediction.confidence >= confVal) &&
191 (prediction.score > scoreVal)) { // data[5] == kilobot score
192 confidences.push_back(prediction.confidence);
193 auto left = int((prediction.x - 0.5 * prediction.width) * xScale);
194 auto top = int((prediction.y - 0.5 * prediction.height) * yScale);
195 auto width = int(prediction.width * xScale);
196 auto height = int(prediction.height * yScale);
197 boxes.emplace_back(left, top, width, height);
198 }
199 }
200 }
201 }
202
203 return YD_RESULT::YD_ERROR;
204 }
